function buildParetoData(){
const all=CATEGORIES.flatMap(cat=>cat.causes.map(c=>({…c,category:cat.name,catColor:cat.color,catBg:cat.bg,score:c.impact*c.freq})));
all.sort((a,b)=>b.score-a.score);
const total=all.reduce((s,c)=>s+c.score,0);
let cum=0;
all.forEach(c=>{cum+=c.score;c.cumPct=Math.round((cum/total)*100);c.pct=Math.round((c.score/total)*100);});
return{items:all,total};
}
